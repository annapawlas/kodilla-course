SELECT I.SUMMARY, I.DESCRIPTION, J.NAME
FROM ISSUES I
JOIN ISSUESLISTS J ON I.ISSUESLIST_ID = J.ID;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(U.ID) AS ISSUES_NUMBER
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID;
